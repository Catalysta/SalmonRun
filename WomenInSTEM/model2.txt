model
		{
			for(i in 1:N) {
				r[i] ~ dbin(p[i],n[i])
				alpha[i] ~ dnorm(0.0,tau)
				logit(p[i]) <- beta0 + beta1 * x1[i] + beta2 * x2[i] + beta3 * x1[i] * x2[i] + alpha[i]
			}
			beta0 ~ dnorm(0.0,0.25)
			beta1 ~ dnorm(0.0,0.25)
			beta2 ~ dnorm(0.0,0.25)
			beta3 ~ dnorm(0.0,0.25)
			sigma ~ dunif(0.0,3.0) 
			tau <- pow(sigma, -2)
		}
